// isInternalIPAddress checks if the host is a loopback, private, or carrier-grade NAT IP address.
func isInternalIPAddress(host string) bool {
	ip := net.ParseIP(host)
	if ip == nil {
		return false
	}

	return ip.IsLoopback() || ip.IsPrivate() || isCarrierGradeNatIP(ip)
}
